Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)
  2. Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass
  3. Parameterized Constructor: ClassName(this.param)
  4. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  5. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  6. ii. Runtime polymorphism: Method overriding (didukung Dart)

abstract class Kendaraan {

  void start();  // Method abstract

class MathUtils {
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Tidak bisa menghitung faktorial bilangan negatif');
    if (n == 0 || n == 1) return 1;
    int hasil = 1;
    for (int i = 2; i <= n; i++) {
      hasil *= i;
    }
    return hasil;
  }
  
  static bool isPrima(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i * i <= n; i += 2) {
      if (n % i == 0) return false;
    }
    return true;
  }
  
  static double roundToDecimal(double value, int decimalPlaces) {
    double faktor = 1;
    for (int i = 0; i < decimalPlaces; i++) {
      faktor *= 10;
    }
    return (value * faktor).round() / faktor;
  }
  
  static int randomInt(int min, int max) {
    return min + (DateTime.now().millisecondsSinceEpoch % (max - min + 1));
  }
  
  static double randomDouble(double min, double max) {
    return min + (DateTime.now().millisecondsSinceEpoch % 10000) / 10000 * (max - min);
  }
  
  static double rataRata(List angka) {
    if (angka.isEmpty) return 0;
    double sum = 0;
    for (var a in angka) {
      sum += a;
    }
    return sum / angka.length;
  }
  
  static double median(List angka) {
    if (angka.isEmpty) return 0;
    List sorted = List.from(angka)..sort();
    int tengah = sorted.length ~/ 2;
    if (sorted.length % 2 == 0) {
      return (sorted[tengah - 1] + sorted[tengah]) / 2;
    } else {
      return sorted[tengah];
    }
  }
}

void main() {
  print('========== SOAL 5 ==========');
  
  print('Factorial 5: ${MathUtils.factorial(5)}');
  print('Factorial 7: ${MathUtils.factorial(7)}');
  print('Apakah 17 prima? ${MathUtils.isPrima(17)}');
  print('Apakah 20 prima? ${MathUtils.isPrima(20)}');
  print('3.14159 ke 2 desimal: ${MathUtils.roundToDecimal(3.14159, 2)}');
  print('Random Int (1-10): ${MathUtils.randomInt(1, 10)}');
  print('Random Double (0-1): ${MathUtils.randomDouble(0, 1)}');
  
  List nilai = [70, 85, 90, 75, 80, 95];
  print('List nilai: $nilai');
  print('Rata-rata: ${MathUtils.rataRata(nilai)}');
  print('Median: ${MathUtils.median(nilai)}');
}

abstract class MediaPlayer {
  String title;
  int duration;
  
  MediaPlayer(this.title, this.duration);
  
  void play();
  void stop();
  void showInfo();
  
  void displayDuration() {
    int menit = duration ~/ 60;

    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
}

// ==================== INTERFACES ====================
abstract interface class Playable {
  void play();
  void resume();
}

abstract interface class Pausable {
  void pause();
  void resumeFromPause();
}

class AudioPlayer implements MediaPlayer, Playable, Pausable {
  @override
  String title;
  
  @override
  int duration;
  
  String formatAudio;
  int bitrate;
  bool isPlaying = false;
  bool isPaused = false;
  
  AudioPlayer({
    required this.title,
    required this.duration,
    required this.formatAudio,
    required this.bitrate,
  });
  
  @override
  void play() {
    if (isPaused) {
      resume();
    } else {
      isPlaying = true;
      isPaused = false;
      print('🎵 Memutar audio: $title ($formatAudio, ${bitrate}kbps)');
    }
  }
  
  @override
  void resume() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('🎵 Melanjutkan audio: $title');
    }
  }
  
  @override
  void stop() {
    isPlaying = false;
    isPaused = false;
    print('⏹️ Menghentikan audio: $title');
  }
  
  @override
  void pause() {
    if (isPlaying) {
      isPlaying = false;
      isPaused = true;
      print('⏸️ Menjeda audio: $title');
    }
  }
  
  @override
  void resumeFromPause() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('▶️ Melanjutkan audio dari jeda: $title');
    }
  }
  
  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📻 AUDIO PLAYER');
    print('═══════════════════════════════');
    print('Judul: $title');
    print('Format: $formatAudio');
    print('Bitrate: ${bitrate}kbps');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
  
  void setVolume(int volume) {
    print('🔊 Mengatur volume ke $volume%');
  }
  
  void setEqualizer(String mode) {
    print('🎚️ Mengatur equalizer ke mode: $mode');
  }
}
